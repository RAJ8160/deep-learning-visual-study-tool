def _decode_hex_pixels(hex_str):
    """Decode 1568-char hex string → (28,28) float32 array in [0,1]."""
    if not hex_str or len(hex_str.strip()) != 1568:
        return None
    try:
        h = hex_str.strip()
        arr = np.array(
            [int(h[i*2:i*2+2], 16) / 255.0 for i in range(784)],
            dtype="float32"
        ).reshape(28, 28)
        return arr
    except Exception:
        return None


def render_draw_pad():
    hex_pixels = st.query_params.get("pixels", None)
    if hex_pixels:
        arr = _decode_hex_pixels(hex_pixels)

@st.cache_data(show_spinner=False)
def load_data():
    csv_train, csv_test = "mnist_train.csv", "mnist_test.csv"
    if os.path.exists(csv_train) and os.path.exists(csv_test):
        df_tr = pd.read_csv(csv_train); df_te = pd.read_csv(csv_test)
        y_tr  = df_tr["label"].values;  y_te  = df_te["label"].values
        X_tr  = df_tr.drop("label", axis=1).values.astype("float32")
        X_te  = df_te.drop("label", axis=1).values.astype("float32")
    else:
        (r_tr, y_tr_r),(r_te, y_te_r) = tf.keras.datasets.mnist.load_data()
        X_tr = np.array(r_tr, dtype="float32", copy=True).reshape(-1, 784)
        X_te = np.array(r_te, dtype="float32", copy=True).reshape(-1, 784)
        y_tr = np.array(y_tr_r, dtype="int32", copy=True)
        y_te = np.array(y_te_r, dtype="int32", copy=True)
    X_tr_flat = np.ascontiguousarray((X_tr / 255.0), dtype="float32")
    X_te_flat = np.ascontiguousarray((X_te / 255.0), dtype="float32")
    X_tr_4d   = np.ascontiguousarray(X_tr_flat.reshape(-1, 28, 28, 1), dtype="float32")
    X_te_4d   = np.ascontiguousarray(X_te_flat.reshape(-1, 28, 28, 1), dtype="float32")
    y_tr = np.array(y_tr, dtype="int32", copy=True)
    y_te = np.array(y_te, dtype="int32", copy=True)
    return X_tr_flat, X_tr_4d, y_tr, X_te_flat, X_te_4d, y_te

PATHS = {
    "perceptron": "perceptron_model.h5",
    "ann":        "ann_model.h5",
    "cnn":        "cnn_model.h5",
}

@st.cache_resource(show_spinner=False)
def load_all_models():
    models = {}
    for key, path in PATHS.items():
        if not os.path.exists(path):
            models[key] = None
            continue
        m = load_model(path, compile=False)
        try:
            raw_shape   = m.input_shape
            dummy_shape = tuple(1 if d is None else d for d in raw_shape)
            dummy       = np.zeros(dummy_shape, dtype="float32")
            m.predict(dummy, verbose=0)
        except Exception:
            for shape in [(1, 28, 28, 1), (1, 28, 28), (1, 784)]:
                try:
                    m.predict(np.zeros(shape, dtype="float32"), verbose=0)
                    break
                except Exception:
                    continue
        models[key] = m
    return models


def get_model_input_shape(model):
    try:
        raw = model.input_shape
        return tuple(1 if d is None else d for d in raw)
    except Exception:
        return (1, 784)


def predict_on_array(model, pixels_28x28: np.ndarray, model_key: str):
    in_shape = get_model_input_shape(model)
    if in_shape[-1] == 1 and len(in_shape) == 4:
        img_in = pixels_28x28.reshape(1, 28, 28, 1)
    elif len(in_shape) == 3:
        img_in = pixels_28x28.reshape(1, 28, 28)
    else:
        img_in = pixels_28x28.reshape(1, 784)
    img_4d = pixels_28x28.reshape(1, 28, 28, 1)
    probs  = model.predict(img_in, verbose=0)[0]
    pred   = int(np.argmax(probs))
    return probs, pred, img_4d

def get_activation_maps(model, img4):
    convs = [l for l in model.layers if isinstance(l, tf.keras.layers.Conv2D)]
    if len(convs) < 2:
        return None
    def to_np(x):
        if hasattr(x, "numpy"): return np.array(x.numpy(), copy=True)
        return np.array(x, copy=True)
    try:
        am = Model(inputs=model.input, outputs=[convs[0].output, convs[1].output])
        c1, c2 = am.predict(img4, verbose=0)
        return {"conv1": to_np(c1), "conv2": to_np(c2)}
    except Exception: pass
    try:
        x = tf.constant(img4, dtype=tf.float32); hit=0; o1=o2=None
        for layer in model.layers:
            x = layer(x, training=False)
            if isinstance(layer, tf.keras.layers.Conv2D):
                hit += 1
                if hit==1: o1=to_np(x)
                elif hit==2: o2=to_np(x); break
        if o1 is not None and o2 is not None:
            return {"conv1": o1, "conv2": o2}
    except Exception: pass
    return None


def activation_thumbnails(act4d, n=16, thumb=8):
    act = act4d[0]; result=[]
    for f in range(min(n, act.shape[-1])):
        fm = act[:,:,f]; mx = fm.max()
        if mx>0: fm = fm/mx
        rows = np.array_split(fm, thumb, axis=0)
        small=[]
        for rg in rows:
            cols = np.array_split(rg, thumb, axis=1)
            small.append([float(round(c.mean(),3)) for c in cols])
        result.append(small)
    return result

# HELPER: get pixel array for animation tabs
# ─────────────────────────────────────────────────────────────────
def get_px28_and_true(prefix: str):
    idx = st.session_state.get(f"{prefix}idx")
    if idx is None:
        return None, None
    if idx == "drawn":
        drawn = st.session_state.get("drawn_pixels")
        if drawn is None:
            return None, None
        return drawn.astype("float32"), None
    return X_te_4d[idx][:,:,0], int(y_test[idx])

# HELPER: render one model animation tab
# ─────────────────────────────────────────────────────────────────
def render_model_tab(model_key, prefix, arch_rows, step_cards,
                     anim_builder, anim_height, comparison_note):
    model = models[model_key]

    if model is None:
        st.markdown(f"""
        <div class='callout cr' style='margin:1rem 0;'>
        ❌  <b>{PATHS[model_key]}</b> not found in the app folder.<br>
        Place the file next to <code>streamlit_app.py</code> and restart.
        </div>""", unsafe_allow_html=True)
        return

    with st.expander("📐 Architecture Reference  (click to expand)", expanded=False):
        for num, name, shape, desc, col in arch_rows:
            st.markdown(f"""
            <div class='lrow'>
              <div class='lnum' style='color:{col};'>{num}</div>
              <div class='lbody'>
                <span class='lname' style='color:{col};'>{name}</span>
                <span class='lshape'>→ {shape}</span>
                <div class='ldesc'>{desc}</div>
              </div>
            </div>""", unsafe_allow_html=True)
        st.markdown(
            f"<p style='color:#64748b;font-size:.82rem;margin-top:.6rem;font-family:monospace;'>"
            f"Total parameters: <span style='color:#4f9eff;'>{model.count_params():,}</span></p>",
            unsafe_allow_html=True)

    idx_now = st.session_state[f"{prefix}idx"]
    pred    = st.session_state[f"{prefix}pred"]

    if idx_now is not None and pred is not None:
        px28, true_label = get_px28_and_true(prefix)
        probs = st.session_state[f"{prefix}probs"]

        if px28 is None:
            st.warning("Drawing data lost — please draw again.")
            return

        vc   = "#34d399" if (true_label is None or pred==true_label) else "#fb923c"
        conf = float(probs[pred]) * 100
        source_str = f"Test #{idx_now}" if isinstance(idx_now, int) else "Your drawing"
        true_str   = f"True: {true_label}" if true_label is not None else "Label: unknown (drawn)"

        st.markdown(f"""
        <div class='callout cb' style='padding:.7rem 1rem;margin-bottom:.6rem;'>
        <div class='metric-row'>
          <div class='metric'><span class='mv' style='color:#4f9eff;'>{source_str}</span><span class='mk'>Source</span></div>
          <div class='metric'><span class='mv' style='color:#fbbf24;'>{true_str}</span><span class='mk'>Label</span></div>
          <div class='metric'><span class='mv' style='color:{vc};'>{pred}</span><span class='mk'>Predicted</span></div>
          <div class='metric'><span class='mv' style='color:{vc};'>{conf:.1f}%</span><span class='mk'>Confidence</span></div>
        </div></div>""", unsafe_allow_html=True)

        step_html = ""
        for emoji, name, col, desc in step_cards:
            step_html += (
                f"<div class='step-mini' style='border-top-color:{col};'>"
                f"<div class='sm-emoji'>{emoji}</div>"
                f"<div class='sm-name' style='color:{col};'>{name}</div>"
                f"<div class='sm-desc'>{desc}</div></div>"
            )
        st.markdown(f"<div style='display:flex;gap:5px;margin:.3rem 0 .7rem;'>{step_html}</div>",
                    unsafe_allow_html=True)

        if model_key == "cnn":
            c1t = st.session_state[f"{prefix}c1t"] or [[[0.5]*8]*8]*16
            c2t = st.session_state[f"{prefix}c2t"] or [[[0.3]*8]*8]*16
            html = anim_builder(px28, true_label, pred, probs, c1t, c2t)
        else:
            html = anim_builder(px28, true_label, pred, probs)

        components.html(html, height=anim_height, scrolling=True)
        st.markdown(comparison_note, unsafe_allow_html=True)
        st.markdown("---")

        st.markdown("### 📊 Detailed Results")
        lc, rc = st.columns([1, 2.2], gap="large")
        with lc:
            title_ov = "Your Drawing" if true_label is None else None
            fig = fig_digit(px28, label=true_label, title_override=title_ov)
            st.pyplot(fig); plt.close(fig)
            st.markdown(f"""
            <div class='rcard' style='border-color:{vc};margin-top:.5rem;'>
              <div class='rdigit' style='color:{vc};'>{pred}</div>
              <div class='rlabel'>Prediction</div>
              <div class='rconf' style='color:{vc};'>{conf:.1f}% confidence</div>
              <div style='margin-top:.5rem;font-size:.88rem;color:{vc};font-weight:600;'>
                {("✅ Correct — True: " + str(true_label)) if true_label is not None and pred==true_label
                  else ("❌ Wrong — True: " + str(true_label)) if true_label is not None
                  else f"🎯 Model says: {pred}"}
              </div>
            </div>""", unsafe_allow_html=True)
        with rc:
            st.pyplot(fig_conf(probs, f"{model_key.upper()} Softmax Output"))
            plt.close()
            top3 = np.argsort(probs)[::-1][:3]
            st.markdown(f"""
            <div class='callout {"cg" if (true_label is None or pred==true_label) else "co"}'>
            <b>Top 3 predictions:</b>&nbsp;
            digit <b>{top3[0]}</b> ({probs[top3[0]]*100:.1f}%)  →
            digit <b>{top3[1]}</b> ({probs[top3[1]]*100:.1f}%)  →
            digit <b>{top3[2]}</b> ({probs[top3[2]]*100:.1f}%)
            </div>""", unsafe_allow_html=True)

        if model_key == "cnn":
            acts = st.session_state[f"{prefix}acts"]
            if acts:
                st.markdown("### 🔬 Real Activation Maps")
                st.markdown("""<div class='callout cp'>
                <b>How to read:</b> Bright/warm tiles = the filter fired strongly.
                Dark tiles = no response. Layer 1 detects edges. Layer 2 detects curves and loops.
                </div>""", unsafe_allow_html=True)
                t1, t2 = st.tabs(["🟣 Conv2D Layer 1 — 32 filters (edges)",
                                   "🔴 Conv2D Layer 2 — 64 filters (curves)"])
                with t1:
                    fig = fig_actmaps(acts["conv1"], "Conv2D Layer 1", n_filters, 8)
                    st.pyplot(fig); plt.close(fig)
                with t2:
                    fig = fig_actmaps(acts["conv2"], "Conv2D Layer 2", n_filters, 8)
                    st.pyplot(fig); plt.close(fig)
    else:
        st.markdown(f"""
        <div style='text-align:center;background:#111827;border:1px solid #1f2d45;
          border-radius:12px;padding:3rem 1rem;'>
          <div style='font-size:3.5rem;'>{"✏️" if mode=="draw" else "🎲"}</div>
          <h3 style='color:#4f9eff!important;margin:.7rem 0 .5rem;'>
            {"Draw a digit above" if mode=="draw" else "Pick a random digit above"}
          </h3>
          <p style='color:#64748b;max-width:420px;margin:0 auto;font-size:.9rem;'>
            The animation will appear here once you provide input.
          </p>
        </div>""", unsafe_allow_html=True)
